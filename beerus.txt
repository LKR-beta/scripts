-- Function to find the nearest player
local function getNearestPlayer()
    local nearestPlayer = nil
    local shortestDistance = math.huge -- Infinite distance to start with

    for _, otherPlayer in pairs(game.Players:GetPlayers()) do
        -- Ignore the local player
        if otherPlayer ~= player and otherPlayer.Character and otherPlayer.Character:FindFirstChild("HumanoidRootPart") then
            -- Check if the other player is in the same game and calculate distance
            local distance = (player.Character.HumanoidRootPart.Position - otherPlayer.Character.HumanoidRootPart.Position).magnitude
            if distance < shortestDistance then
                shortestDistance = distance
                nearestPlayer = otherPlayer
            end
        end
    end

    return nearestPlayer
end

-- On activation (when clicked)
tool.Activated:Connect(function()
    -- OPTIONAL: Play animation or sound here if needed

    -- Ensure player character is loaded
    if player.Character and player.Character:FindFirstChild("HumanoidRootPart") then
        -- Teleport to the nearest player
        local nearestPlayer = getNearestPlayer()
        if nearestPlayer and nearestPlayer.Character then
            -- Ensure the nearest player also has a loaded character and HumanoidRootPart
            if nearestPlayer.Character:FindFirstChild("HumanoidRootPart") then
                local nearestHumanoidRootPart = nearestPlayer.Character.HumanoidRootPart

                -- Save the current position of the local player (for teleport back)
                local originalPosition = player.Character.HumanoidRootPart.Position

                -- Calculate position behind the nearest player (e.g., offset by -5 in the Z direction)
                local punchPosition = nearestHumanoidRootPart.Position - nearestHumanoidRootPart.CFrame.LookVector * 5
                
                -- Teleport the local player to the punch position (behind the nearest player)
                player.Character:SetPrimaryPartCFrame(CFrame.new(punchPosition))

                -- Now, after the punch (activate punch), we will teleport the player back
                local function getOriginalPunch()
                    return player.Backpack:FindFirstChild("Normal Punch") or player.Character:FindFirstChild("Normal Punch")
                end

                local original = getOriginalPunch()

                if original and original:IsA("Tool") then
                    -- Equip and activate the original punch
                    if original.Parent == player.Backpack then
                        original.Parent = player.Character
                        task.wait(0.1)
                    end
                    original:Activate()

                    
                    task.wait(1) 
                    player.Character:SetPrimaryPartCFrame(CFrame.new(originalPosition))
                else
                    warn("Original 'Normal Punch' tool not found!")
                end
            else
                warn("Nearest player has no HumanoidRootPart")
            end
        else
            warn("No nearby player found!")
        end
    else
        warn("Local player character is not fully loaded or doesn't have a HumanoidRootPart")
    end
end)

local AccessorySettings = {
    RemoveOriginalAccessories = true --set true to remove accessories
}

local function DisableCollisions(object)
    if object:IsA("BasePart") then
        object.CanCollide = false
        object.CanTouch = false
        object.CanQuery = false
    end
end

local function RemoveObjectAfterDelay(object)
    -- Destroy the object after 2 seconds
    if object then
        delay(2, function()
            if object.Parent then
                object:Destroy()  -- Destroy the object after the delay
            end
        end)
    end
end

local function AttachAccessories(accessoryTable)
    local player = game.Players.LocalPlayer
    local character = player.Character or player.CharacterAdded:Wait()

    if AccessorySettings.RemoveOriginalAccessories then
        for _, obj in pairs(character:GetChildren()) do
            if obj:IsA("Accessory") then
                obj:Destroy()
            end
        end
    end

    for _, accessoryData in pairs(accessoryTable) do
        local item = game:GetObjects("rbxassetid://" .. tostring(accessoryData.AssetID))[1]
        if not item then
            warn("Failed to load item with ID: " .. tostring(accessoryData.AssetID))
            return
        end

        --convert
        if item:IsA("Tool") then
            local newAccessory = Instance.new("Accessory")
            local handle = item:FindFirstChildWhichIsA("Part") or item:FindFirstChildWhichIsA("MeshPart")

            if handle then
                handle.Parent = newAccessory
                newAccessory.Parent = character
                newAccessory.Name = "ConvertedAccessory"

                DisableCollisions(handle)

                -- Remove object after 2 seconds
                RemoveObjectAfterDelay(handle)

                local targetPart = character:FindFirstChild(accessoryData.AttachTo)
                if targetPart then
                    local weld = Instance.new("Weld")
                    weld.Part0 = targetPart
                    weld.Part1 = handle
                    weld.C0 = CFrame.new(accessoryData.Position) * CFrame.Angles(
                        math.rad(accessoryData.Rotation.X),
                        math.rad(accessoryData.Rotation.Y),
                        math.rad(accessoryData.Rotation.Z)
                    )
                    weld.Parent = targetPart
                end
            end
        else
            item.Parent = character
            local targetPart = character:FindFirstChild(accessoryData.AttachTo)
            if targetPart then
                local attachPart = item:FindFirstChildWhichIsA("MeshPart") or item:FindFirstChildWhichIsA("Part")
                if attachPart then
                    DisableCollisions(attachPart)

                    -- Remove object after 2 seconds
                    RemoveObjectAfterDelay(attachPart)

                    local weld = Instance.new("Weld")
                    weld.Part0 = targetPart
                    weld.Part1 = attachPart
                    weld.C0 = CFrame.new(accessoryData.Position) * CFrame.Angles(
                        math.rad(accessoryData.Rotation.X),
                        math.rad(accessoryData.Rotation.Y),
                        math.rad(accessoryData.Rotation.Z)
                    )
                    weld.Parent = targetPart
                end
            end
        end
    end
end

local function getNearestPlayer()
    local nearestPlayer = nil
    local shortestDistance = math.huge -- Infinite distance to start with

    for _, otherPlayer in pairs(game.Players:GetPlayers()) do
        -- Ignore the local player
        if otherPlayer ~= player and otherPlayer.Character and otherPlayer.Character:FindFirstChild("HumanoidRootPart") then
            -- Check if the other player is in the same game and calculate distance
            local distance = (player.Character.HumanoidRootPart.Position - otherPlayer.Character.HumanoidRootPart.Position).magnitude
            if distance < shortestDistance then
                shortestDistance = distance
                nearestPlayer = otherPlayer
            end
        end
    end

    return nearestPlayer
end

local AccessorySettings = {
    RemoveOriginalAccessories = true --set true to remove accessories
}

local function DisableCollisions(object)
    if object:IsA("BasePart") then
        object.CanCollide = false
        object.CanTouch = false
        object.CanQuery = false
    end
end

local function RemoveObjectAfterDelay(object)
    -- Destroy the object after 2 seconds
    if object then
        delay(2, function()
            if object.Parent then
                object:Destroy()  -- Destroy the object after the delay
            end
        end)
    end
end

local function AttachAccessories(accessoryTable)
    local player = game.Players.LocalPlayer
    local character = player.Character or player.CharacterAdded:Wait()

    if AccessorySettings.RemoveOriginalAccessories then
        for _, obj in pairs(character:GetChildren()) do
            if obj:IsA("Accessory") then
                obj:Destroy()
            end
        end
    end

    for _, accessoryData in pairs(accessoryTable) do
        local item = game:GetObjects("rbxassetid://" .. tostring(accessoryData.AssetID))[1]
        if not item then
            warn("Failed to load item with ID: " .. tostring(accessoryData.AssetID))
            return
        end

        --convert
        if item:IsA("Tool") then
            local newAccessory = Instance.new("Accessory")
            local handle = item:FindFirstChildWhichIsA("Part") or item:FindFirstChildWhichIsA("MeshPart")

            if handle then
                handle.Parent = newAccessory
                newAccessory.Parent = character
                newAccessory.Name = "ConvertedAccessory"

                DisableCollisions(handle)

                -- Remove object after 2 seconds
                RemoveObjectAfterDelay(handle)

                local targetPart = character:FindFirstChild(accessoryData.AttachTo)
                if targetPart then
                    local weld = Instance.new("Weld")
                    weld.Part0 = targetPart
                    weld.Part1 = handle
                    weld.C0 = CFrame.new(accessoryData.Position) * CFrame.Angles(
                        math.rad(accessoryData.Rotation.X),
                        math.rad(accessoryData.Rotation.Y),
                        math.rad(accessoryData.Rotation.Z)
                    )
                    weld.Parent = targetPart
                end
            end
        else
            item.Parent = character
            local targetPart = character:FindFirstChild(accessoryData.AttachTo)
            if targetPart then
                local attachPart = item:FindFirstChildWhichIsA("MeshPart") or item:FindFirstChildWhichIsA("Part")
                if attachPart then
                    DisableCollisions(attachPart)

                    -- Remove object after 2 seconds
                    RemoveObjectAfterDelay(attachPart)

                    local weld = Instance.new("Weld")
                    weld.Part0 = targetPart
                    weld.Part1 = attachPart
                    weld.C0 = CFrame.new(accessoryData.Position) * CFrame.Angles(
                        math.rad(accessoryData.Rotation.X),
                        math.rad(accessoryData.Rotation.Y),
                        math.rad(accessoryData.Rotation.Z)
                    )
                    weld.Parent = targetPart
                end
            end
        end
    end
end
